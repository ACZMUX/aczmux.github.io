using namespace std;

	int a,b,c;
	cin>>a>>b;
	c=a;
	for(int i=1;i<b;i++){
		c=c*a;
		//cout<<"[Debug] c="<<c<<endl;
		c=c%1000;
		//cout<<"[Debug] after move c="<<c<<endl;

	cout<<c<<endl;
	return 0;
}

最后一个测试点没过，想起题面【输出】部分的要求：“若幂本身不足三位，在前面补零。”
那么怎么判断它的位数呢？字符串相关的内容用起来比较麻烦也不想回去翻书了，于是就用比大小的方法判断，并确定输出方式。

#include<bits/stdc++.h>
using namespace std;
int main(){
	int a,b,c;
	cin>>a>>b;
	c=a;
	for(int i=1;i<b;i++){
		c=c*a;
		//cout<<"[Debug] c="<<c<<endl;
		c=c%1000;
		//cout<<"[Debug] after move c="<<c<<endl;
	} 
	if(c<100){
		if(c<10){
			cout<<"00"<<c<<endl; 
			return 0;
		}
		cout<<"0"<<c<<endl;
	}
	cout<<c<<endl;
	return 0;
}
